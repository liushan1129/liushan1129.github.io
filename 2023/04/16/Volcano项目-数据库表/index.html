Volcano项目-数据库表

John Doe
2023-04-16

用户表
CREATE TABLE va_user_basic (
id bigint unsigned NOT NULL AUTO_INCREMENT COMMENT ‘主键’,
role_ids varchar(200) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL DEFAULT ‘’ COMMENT ‘角色主键’,
dept_ids varchar(200) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL DEFAULT ‘’ COMMENT ‘部门ID’,
post_ids varchar(200) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL DEFAULT ‘’ COMMENT ‘职位ID’,
username varchar(32) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL DEFAULT ‘’ COMMENT ‘用户账号’,
nickname varchar(32) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL DEFAULT ‘’ COMMENT ‘用户昵称’,
avatar varchar(200) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL DEFAULT ‘’ COMMENT ‘用户头像’,
gender tinyint unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘性别: 0=女, 1=男’,
age int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘年龄’,
weight int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘体重’,
height int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘身高’,
telephone varchar(20) DEFAULT NULL COMMENT “电话号码”,
sort smallint unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘排序编号’,
password varchar(200) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL DEFAULT ‘’ COMMENT ‘用户密码’,
salt varchar(20) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL DEFAULT ‘’ COMMENT ‘加密盐巴’,
is_multipoint tinyint unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘多端登录: 0=否, 1=是’,
is_disable tinyint unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘是否禁用: 0=否, 1=是’,
is_delete tinyint unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘是否删除: 0=否, 1=是’,
last_login_ip varchar(20) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL DEFAULT ‘’ COMMENT ‘最后登录IP’,
last_login_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘最后登录’,
create_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘创建时间’,
update_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘更新时间’,
delete_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘删除时间’,
PRIMARY KEY (id) USING BTREE
) ENGINE=InnoDB AUTO_INCREMENT=9 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci ROW_FORMAT=DYNAMIC COMMENT=’用户表’

会员表
CREATE TABLE va_member_basic (
id bigint unsigned NOT NULL AUTO_INCREMENT COMMENT ‘主键’,
name varchar(32) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL DEFAULT ‘’ COMMENT ‘姓名’,
gender tinyint unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘性别: 0=女, 1=男’,
age int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘年龄’,
weight int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘体重’,
height int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘身高’,
telephone varchar(20) DEFAULT NULL COMMENT “电话号码”,
join_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘加入时间’,
course_type_id bigint NOT NULL DEFAULT ‘0’ COMMENT ‘课程类型id’,
is_end tinyint unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘是否结课: 0=否, 1=是’,
is_delete tinyint unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘是否删除: 0=否, 1=是’,
expire_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘过期时间’,
create_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘创建时间’,
update_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘更新时间’,
delete_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘删除时间’,
PRIMARY KEY (id) USING BTREE
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci ROW_FORMAT=DYNAMIC COMMENT=’会员表’

课程类型表
CREATE TABLE va_course_type (
id bigint unsigned NOT NULL AUTO_INCREMENT COMMENT ‘主键’,
name varchar(32) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL DEFAULT ‘’ COMMENT ‘课程名称’,
type bigint NOT NULL COMMENT ‘课程类型’,
times int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘课时’,
is_delete tinyint unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘是否删除: 0=否, 1=是’,
create_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘创建时间’,
update_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘更新时间’,
delete_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘删除时间’,
PRIMARY KEY (id) USING BTREE
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci ROW_FORMAT=DYNAMIC COMMENT=’课程类型表’

小组表
CREATE TABLE va_group (
id bigint unsigned NOT NULL AUTO_INCREMENT COMMENT ‘主键’,
name varchar(32) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL DEFAULT ‘’ COMMENT ‘课程名称’,
user_id bigint NOT NULL DEFAULT ‘0’ COMMENT ‘用户id’,
username varchar(32) DEFAULT ‘’ COMMENT ‘用户账号’
member_ids varchar(200) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NOT NULL DEFAULT ‘’ COMMENT ‘会员id集合’,
member_names varchar(255) DEFAULT ‘’ COMMENT ‘会员名称’
is_delete tinyint unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘是否删除: 0=否, 1=是’,
create_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘创建时间’,
update_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘更新时间’,
delete_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘删除时间’,
PRIMARY KEY (id) USING BTREE
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci ROW_FORMAT=DYNAMIC COMMENT=’小组表’

预约记录表
CREATE TABLE va_reservation_record (
id bigint unsigned NOT NULL AUTO_INCREMENT COMMENT ‘主键’,
group_id bigint NULL COMMENT ‘小组id’,
user_id bigint NOT NULL DEFAULT ‘0’ COMMENT ‘用户id’,
member_ids varchar(200) NOT NULL DEFAULT ‘’ COMMENT ‘会员id集合’,
reservation_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘预约时间时间’,
status tinyint unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘状态: 0=完成, 1=取消’,
create_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘创建时间’,
update_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘更新时间’,
cancel_time int unsigned NOT NULL DEFAULT ‘0’ COMMENT ‘取消时间’,
PRIMARY KEY (id) USING BTREE
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci ROW_FORMAT=DYNAMIC COMMENT=’预约记录表’
